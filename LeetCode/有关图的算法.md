# 有关图的算法

## 深度优先搜索 dfs
示例输入：
```
7 5  //图的节点个数，连接个数
1 2
2 3
2 4
5 6
5 7
```

要求输出：
```
1 2 3 4
5 6 7
```

C++代码实现
```cpp
/*
图的深度优先遍历算法 dfs
用邻接表来实现
*/
/*
简单版的基于邻接表的dfs
*/

#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

const int MAXN = 10010;

vector<int> adjLink[MAXN];  //邻接表
int visited[MAXN]; //记录哪个节点被访问

void dfs(int u){
    visited[u] = 1;
    cout << u << " ";
    for (int i=0; i<adjLink[u].size(); i++){
        int v = adjLink[u][i];
        if (!visited[v])  dfs(v);
    }
}

int main(){
    int n, m;  //节点个数，连接个数
    cin >> n >> m;

    for (int i=0; i<m; i++){
        int x, y;
        cin >> x >> y;
        adjLink[x].push_back(y);
    }

    for (int i=1; i<=n; i++){
        if (!visited[i]){
            dfs(i);
            cout << endl;
        }
    }

    return 0;
}
```

## 求有向图的强连通分量  Tarjan算法
参考思路解析：[trajan算法](https://iq.opengenus.org/tarjans-algorithm/)

示例：
```
输入样例：
8 9  // 节点个数，连接个数
1 3  // 每条边的起始点和结束点
2 1
3 2
3 4
4 5
5 6
6 7
7 8
8 5
输出样例：
low: 1 1 1 4 5 5 5 5
dfn: 1 3 2 4 5 6 7 8
scc: 3 3 3 2 1 1 1 1   // 每个元素对应的强连通分量的标号
subnum: 4 1 3   // 不同子集包含的元素个数
```
C++ 代码实现：
```cpp
#include<cstdio>
#include<iostream>
#include<cstring>
#include<vector>
#include<stack>
using namespace std;

const int MAXN = 10010;

int dfn[MAXN],low[MAXN];
int p,group_index;
int scc[MAXN],subnum[MAXN];
vector<int> G[MAXN]; //邻接表
bool v1[MAXN];
stack<int> s;

void Tarjan(int u)
{
	p++;
	low[u]=p;
	dfn[u]=p;
	v1[u]=1;
	s.push(u);
	for(int i=0;i<G[u].size();i++)
	{
		int v=G[u][i];
		if(!dfn[v])
		{
			Tarjan(v);
			low[u]=min(low[u],low[v]);
		}
		else if(v1[v])
			low[u]=min(low[u],dfn[v]);
	}
	if(low[u]==dfn[u])
	{
		int p;
		group_index++;
		do
		{
			p=s.top();
			v1[p]=0;
			scc[p]=group_index;
            subnum[group_index] += 1;
			s.pop();
		}while(p!=u);
	}
}

int main(){
    int n, m;
	cin >> n >> m;
    for (int i=0; i<m; i++){
        int tmpx, tmpy;
		cin >> tmpx >> tmpy;
        G[tmpx].push_back(tmpy);
    }
    for(int i=1;i<=n;i++){
        if(!dfn[i])
            Tarjan(i);
    }

    cout << "low: ";
    for (int i=1; i<=n; i++) cout << low[i] << " ";
    cout << endl;

    cout << "dfn: ";
    for (int i=1; i<=n; i++)  cout << dfn[i] << " ";
    cout << endl;

    cout << "scc: ";
    for (int i=1; i<=n; i++)  cout << scc[i] << " ";
    cout << endl;

    cout << "subnum: ";
    for (int i=1; i<=group_index; i++) cout << subnum[i] << " ";
    cout << endl;
        
    return 0;
}
```
